Why Do We Use AngularJS?

What's Not So Good About It?

Documentation :(((
Few community plugins.
Best practices are still emerging.
There is a learning curve.
No deployment solution for large apps.

What's So Good About It Then?

Encourages separation of concerns.
Dependency injection.
Testing is EASY!
Don't need to write DOM glue.
Declarative like HTML and CSS.

Separation of Concerns


angular.controller('MyController', function($scope){
	// View logic
});
						

angular.factory('MyService', function(){
	// Reusable logic
});
						

angular.directive('MyDirective', function(){
	// DOM logic
});

Dependency Injection


angular.factory('BiznizLogic', function(){
	return function(a, b) {
		return a + b;
	};
});
angular.controller('MyController', function($scope, BiznizLogic){
	$scope.aValue = BiznizLogic(1, 2);
});

Easy Testing


describe('MyController', function(){

	it('should call BiznizLogic', inject(function($controller){
		var spy = jasmine.createSpy('BiznizLogic spy');
		$controller('MyController', { BiznizLogic:  spy});
		expect(spy).toHaveBeenCalled();
	}));

});

No DOM Glue

{{ input }}


<div ng-controller="NoDomGlue">
	<input type="text" ng-model="input">
	<p>{{ input }}</p>
</div>
						

angular.controller('NoDomGlue', function($scope) {
	$scope.input = "Initial value";
});

Declarative

Toggle Pane My Pane 

<div ng-controller="Declarative">
	<a ng-click="togglePane()">Toggle Pane</a>
	<pane ng-show="isPaneVisible">My Pane</pane>
</div>
						

angular.directive('pane', function() {
	return {
		restrict: 'E',
		template: '<div class="pane" ng-transclude> \
			   <div class="pane-inner" ng-transclude> \
			   </div></div>',
		replace: true,
		transclude: true
	};
});
